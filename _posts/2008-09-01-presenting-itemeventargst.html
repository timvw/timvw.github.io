Because i believe in the DRY</a> principle i decided to add a generic ItemEventArgs class to BeTimvwFramework</a>:</p>

[code lang="csharp"]public class ItemEventArgs : EventArgs
{
private T item;

public ItemEventArgs(T item)
{
this.item = item;
}

public T Item
{
get { return this.item; }
set { this.item = value; }
}
}[/code]
