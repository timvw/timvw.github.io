Being bored of writing code to raise an event, i have added an EventHandlerHelper to BeTimvwFramework</a>.</p>
[code lang="csharp"]public static class EventHandlerHelper
{
[MethodImpl(MethodImplOptions.NoInlining)]
public static void Raise(EventHandler handler, object sender, T e)
where T : EventArgs
{
if (handler != null)
{
handler(sender, e);
}
}
}[/code]
