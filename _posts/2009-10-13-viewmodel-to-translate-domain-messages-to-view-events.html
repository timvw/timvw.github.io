Here is an example of a ViewModel that translates domain messages to view events:</p>

[code lang="csharp"]class GameViewModel : INotifyPropertyChanged, IListener
{
public event PropertyChangedEventHandler PropertyChanged = delegate { };

public GameViewModel()
{
var messageBus = ServiceLocator.MessageBus;
messageBus.Subscribe(this);
}

void IListener.Handle(BoardChanged message)
{
PropertyChanged("Board");
}
}[/code]
